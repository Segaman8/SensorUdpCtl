156 void *threadRoutine (void *)
157 {
158  /* vars */
159  int readStatus;
160  struct sockaddr_in serAddr, cliAddr;
161  socklen_t cliAddrLen = sizeof (cliAddr);
162  char buff[1024] = {0};
163 
164  /* creating a new server socket */
165  if ((serSockDes = socket (AF_INET, SOCK_DGRAM, 0)) < 0)
166  {
167  perror ("socket creation error...\n");
168  exit (-1);
169  }
170 
171  /* binding the port to ip and port */
172  serAddr.sin_family = AF_INET;
173  serAddr.sin_port = htons (7989);
174  serAddr.sin_addr.s_addr =
175  //inet_addr ("192.168.0.255");
176  INADDR_ANY;
177 
178  /* bind with trycatch */
179  try
180  {
181  if ((bind (serSockDes, (struct sockaddr *)&serAddr, sizeof (serAddr))) < 0)
182  {
183  Operation()()->setBindState (false);
184  close (serSockDes);
185  return nullptr;
186  }
187  else
188  Operation()()->setBindState (true);
189  }
190  catch (const std::exception &ex)
191  {
192  except (ex);
193  close (serSockDes);
194  return 0;
195  }
196 
197  /* read via cycle */
198  while (true)
199  {
200  /* get msg */
201  readStatus =
202  recvfrom (
203  serSockDes,
204  buff, 1024, 0,
205  (struct sockaddr *)&cliAddr,
206  &cliAddrLen
207  );
208 
209  try
210  {
211  /* got error */
212  if (readStatus < 0)
213  {
214  Operation()()->setBindState (false);
215  close (serSockDes);
216  return nullptr;
217  }
218 
219  /* finalize */
220  buff[readStatus] = '\0';
221 
222  {
223  /* begin operation */
224  Operation op;
225 
226  /* increase counter */
227  op()->incMsgCounter();
228 
229  /* parse and store */
230  op += DataHelper (buff, readStatus);
231  }
232  }
233  catch (const std::exception &ex)
234  {
235  except (ex);
236  break;
237  }
238  }
239 
240  /* finish */
241  close (serSockDes);
242 
243  return 0;
244 }
